        function hitungIMT() {
            const berat = parseFloat(document.getElementById("berat").value);
            const tinggi = parseFloat(document.getElementById("tinggi").value) / 100;
            if (!berat || !tinggi) return alert("Mohon isi berat dan tinggi dengan benar.");
            const imt = berat / (tinggi * tinggi);
            let kategori = "";
            if (imt < 18.5) kategori = "Kurus";
            else if (imt < 25) kategori = "Normal";
            else if (imt < 30) kategori = "Gemuk";
            else kategori = "Obesitas";
            document.getElementById("hasil-imt").innerText = `IMT Anda: ${imt.toFixed(2)} (${kategori})`;
        }

        function hitungBBI() {
            const tinggi = parseFloat(document.getElementById("tinggi-bbi").value);
            if (!tinggi) return alert("Mohon isi tinggi badan dengan benar.");
            const bbi = (tinggi - 100) - ((tinggi - 100) * 0.1);
            document.getElementById("hasil-bbi").innerText = `Berat Badan Ideal Anda: ${bbi.toFixed(1)} kg`;
        }

        const rekomendasi = {
            anak: ["Susu tinggi kalsium", "Sayuran berwarna", "Buah segar", "Telur", "Ikan laut"],
            remaja: ["Sereal penuh serat", "Protein tinggi (ayam, tempe)", "Sayur dan buah", "Air putih cukup"],
            dewasa: ["Karbohidrat kompleks (nasi merah)", "Sayur hijau", "Ikan", "Buah rendah gula"],
            lansia: ["Bubur atau makanan lunak", "Sayuran rebus", "Buah potong", "Susu rendah lemak"],
            hamil: ["Asam folat (bayam, kacang)", "Protein tinggi", "Buah-buahan segar", "Susu ibu hamil"]
        };

        function tampilkanMakanan(kelompok) {
            const list = document.getElementById("daftar-makanan");
            list.innerHTML = "";
            if (rekomendasi[kelompok]) {
                rekomendasi[kelompok].forEach(makanan => {
                    const li = document.createElement("li");
                    li.textContent = makanan;
                    list.appendChild(li);
                });
            }
        }
